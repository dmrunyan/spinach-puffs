\begin{recipe}
    [
      portion = {\portion{5-6}}
    ]
    {Farfalle with Sausage}



  \ingredients[10]{
    & farfalle pasta \\
    & olive oil \\
    \unit[12]{oz} & sweet Italian sausage \\
    & crushed red pepper \\
    1 & diced white onion \\
    1 bulb & garlic (minced) \\
    1 can & Italian-style plum tomatoes \\
    \unit[1.5]{cups} & heavy cream \\
    & salt \\
    & parsley
  }

  \preparation{
    \step Bring a large pot of lightly salted water to a boil. Cook pasta in water until al dente. (move on to other steps while this is going on)
    \step Heat oil in a large, deep skillet over medium heat. Cook sausage and pepper flakes until sausage is evenly brown.
    \step Stir in onion and garlic, and cook until onion is tender.
    \step Stir in tomatos, cream, and salt. Simmer until mixture thickens.
    \step Stir cooked pasta into sauce and heat through. Sprinkle with parsley.
  }
  

\end{recipe}
