  \begin{recipe}
  [%
    portion = {\portion{5-6}}
  ]
  {Beef Stroganoff}
    \ingredients[13]{%
      \unit[2]{lb}                & sirloin steak \\
      \unit[2]{Tbsp}              & olive oil \\
      \unit[2]{Tbsp}              & butter \\
      1                           & medium onion (sliced) \\
      \unit[8]{oz}                & white mushrooms (sliced) \\
      \unit[3]{cloves}            & garlic (minced) \\
      \unit[\nicefrac{1}{2}]{cup} & brandy \\
      \unit[14]{oz can}           & beef broth \\
      \unit[14]{oz can}           & cream of mushroom soup \\
      \unit[1]{cup}               & sour cream \\
      \unit[16]{oz}               & egg noodles \\
      to taste                    & salt \\
      to taste                    & black pepper
    }
    \preparation{%
      \step Cut the meat into bite-sized cubes.
      \step In a large, stainless steel skillet, brown the meat in the olive oil and butter
            with a pinch of salt over medium heat. Remove the meat, but leave the drippings.
      \step Saut\'{e} the onion and muchrooms in the skillet until the vegetables begin to soften.
      \step Add the minced garlic and saut\'{e} for another 2 minutes.
      \step Sprinkle the flour into the skillet and cook for another 3 minutes, stirring constantly.
      \step Remove the vegetables from the skillet and deglaze with the brandy.
      \step Add the beef broth, mushroom soup, meat, vegetables, salt, and pepper to the skillet.
            Simmer over low heat for 30 minutes, stirring occasionally.
      \step While the sauce is simmering, bring a pot of water to boil, then add the egg noodles.
            Cook until tender, then drain.
      \step Temper the sour cream with some of the sauce, then stir the tempered mixture back into the pan.
      \step Serve the sauce over the egg noodles.
    }
  \end{recipe}
