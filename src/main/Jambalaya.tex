  \begin{recipe}
  [%
    portion = {\portion{6-8}}
  ]
  {Jambalaya}
    \ingredients[23]{%
      \unit[1]{Tbsp}                & cooking oil \\
      1                             & onion (chopped) \\
      1                             & green bell pepper (chopped) \\
      \unit[2]{ribs}                & celery (chopped) \\
      \unit[1]{lb}                  & chicken (diced) \\
      \unit[12]{oz}                 & andoulle sausage (medallioned) \\
      \unit[\nicefrac{1}{2}]{cup}   & butter \\
      \unit[\nicefrac{1}{2}]{cup}   & flour \\
      \unit[5]{cans}                & chicken broth \\
      \unit[1]{can}                 & diced tomatoes \\
      \unit[1]{Tbsp}                & chili powder \\
      \unit[3]{Tbsp}                & cumin \\
      \unit[1]{tsp}                 & poultry seasoning \\
      \unit[1]{Tbsp}                & garlic powder \\
      \unit[1]{tsp}                 & cayenne pepper \\
      \unit[\nicefrac{1}{2}]{tsp}   & paprika \\
      \unit[\nicefrac{1}{2}]{tsp}   & ground black pepper \\
      \unit[\nicefrac{1}{2}]{tsp}   & ground white pepper \\
      \unit[\nicefrac{1}{2}]{tsp}   & thyme \\
      to taste                      & salt \\
      \unit[1]{cup}                 & Southern Comfort \\
      \unit[2\nicefrac{1}{2}]{cups} & long grain brown rice \\
      \unit[1]{lb}                  & shrimp
    }
    \preparation{%
      \step Saut\'{e} onion, green pepper, and celery in oil with a pinch of salt
            over medium heat until tender, then set aside.
      \step Brown chicken and sausage, then remove from pot and set aside as well.
      \step Make a roux with the butter and flour. (Melt butter over medium heat, then
            add flour. Stir constantly until it is the color of peanut butter. \unit[~20]{mins})
      \step Add the chicken broth, tomatoes, spices, Southern Comfort, meat, and vegetables,
            then bring to a boil.
      \step Reduce heat, cover, and let simmer for 15-20 minutes.
      \step Add rice, cover, and continue slow simmer for 40 minutes.
      \step Add shrimp, cover, and simmer another 15 minutes.
    }
  \end{recipe}
