  \begin{recipe}
  [%
    portion = {\portion{4}}
  ]
  {Cheesy Cauliflower}
    \ingredients[11]{%
      \unit[1]{Tbsp}              & butter \\
      \unit[1]{head}              & cauliflower \\
      \unit[5.5]{oz}              & grated cheddar \\
      \unit[\nicefrac{1}{2}]{cup} & coarse bread crumbs \\
      1                           & egg \\
      \unit[1]{tsp}               & salt \\
      \unit[1]{tsp}               & mustard powder \\
      \unit[\nicefrac{1}{2}]{tsp} & paprika \\
      \unit[\nicefrac{1}{4}]{tsp} & ground plack pepper \\
      \unit[\nicefrac{1}{4}]{tsp} & cayenne pepper \\
      \unit[\nicefrac{1}{2}]{cup} & heavy cream
    }
    \preparation{%
      \step Grease glass casserole dish with the butter and set aside.
      \step Position one oven rack in the middle, and one on the top of the oven.
            Preheat to \unit[400]{\degree F}
      \step Place cauliflower in a glass bowl, stalk end up.
      \step Use a pairing knife to remove the lower leaves.
      \step Trim the curds off of the stem, keeping them as intact as possible.
            Try to keep the size uniflorm and make as few cuts as possible.
            Discard the stem and the leaves.
      \step Cover tightly with plastic wrap and microwave on high for \unit[4]{mins}.
      \step Rest, still covered in plastic, for another \unit[4]{mins}.
      \step Uncover and rest for another \unit[4-5]{mins} more.
      \step Combine \unit[4.5]{oz} of the cheese, the bread crumbs, egg, salt,
            mustard, paprika, black pepper, and cayenne in a large mixing bowl.
      \step When 1 minute remains on the cauliflower resitn period, microwave the
            heavy cream on high for 30 seconds.
      \step Add the cream to the bowl with the cheese mixture and stir to combine.
      \step Using your hands and a tea towel if the cauliflower is still hot, rub the
            large florets between your palms to break up the curds.
      \step Add the curds to the cream mixture and stir until the cheese
            is melted and combined.
      \step Spread the cauliflower mixture in the prepared baking dish and bake
            for 20 minutes on the middle rack.
      \step Remove the baking dish and turn the broiler to high.
      \step Sprinkle the remaining \unit[1]{oz} cheese on top of the cauliflower.
      \step Place on the top rack and broil until the top is bubbly and slightly browned.
            \unit[6-8]{mins}.
      \step Rest 10 minutes before serving.
    }
  \end{recipe}
